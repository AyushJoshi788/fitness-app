## Data Models

### User Document Structure
```typescript
/users/{userId}
{
  userId: string                  // Firebase Auth UID
  email: string                   // user@example.com
  displayName: string             // John Doe
  height: number                  // cm
  weight: number                  // kg
  experienceLevel: 'beginner' | 'intermediate' | 'advanced'
  subscription: {
    isPremium: boolean
    startDate: timestamp
    endDate: timestamp
  }
  preferences: {
    // User preferences
  }
  createdAt: timestamp
  updatedAt: timestamp
}
```

### Workout Document Structure
```typescript
/users/{userId}/workouts/{workoutId}
{
  id: string                      // unique ID
  userId: string                  // owner
  date: timestamp                 // workout date
  type: string                    // 'cardio', 'strength', etc
  exercises: [
    {
      name: string
      category: string
      muscleGroups: string[]
      sets: number
      reps: number
      weight?: number
      difficulty: string
    }
  ]
  duration: number                // minutes
  notes: string                   // user notes
  createdAt: timestamp
}
```

### Audit Result Document Structure
```typescript
/users/{userId}/audits/{auditId}
{
  workoutId: string
  date: timestamp
  score: number                   // 0-100
  findings: [
    {
      rule: string
      severity: 'info' | 'warning' | 'critical'
      message: string
      recommendation?: string
    }
  ]
  createdAt: timestamp
}
```
